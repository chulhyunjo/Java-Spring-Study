```java
import java.io.*;

// 집 주소
public class Main {
    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        while (true) {
            char[] cArr = br.readLine().toCharArray();
            if (cArr[0] == '0') {
                break;
            }else{
                int calV = 1;
                for (char c : cArr) {
                    if (c == '1') {
                        calV += 3;
                    } else if (c == '0') {
                        calV += 5;
                    } else {
                        calV += 4;
                    }
                }
                System.out.println(calV);
            }
        }
        br.close();
    }
}
```
